      # Key Vault

      # Create Key Vault

      az keyvault create \
    --resource-group <resource-group> \
    --name <your-unique-vault-name>

      # Show a Secret from a Key Vault

      az keyvault show \
  --name <secret name> \
  --vault-name <key vault>
